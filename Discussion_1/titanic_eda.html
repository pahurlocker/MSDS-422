fig = plt.figure(figsize=(14,6))
gs = GridSpec(2, 2, figure=fig)

ax1 = fig.add_subplot(gs[0, :])
df_train.Age.hist(ax=ax1)
ax1.set_title("Age Distribution")
ax2 = fig.add_subplot(gs[-1, 0])
